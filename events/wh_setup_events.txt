					martial = 10
					trait = misguided_warrior

				}
			}
			add_trait = bretonnian_knight

		}
		if = {
			limit = {
				OR = {
					trait = bretonnian_knight

				}
				NOT = {
					has_minor_title = title_sir_bretonnian

				}
			}

					is_female = yes
				}
				give_minor_title = title_lady_knight_bretonnian
			}
		}
		# set_global_flag = knights_set
	}

	}
