# Health Events (just Warhammer diseases)
# Id : diseases namespace
namespace = disease

# Tomb Rot
character_event = {
	id = disease.1
	desc = EVTDESC_DISEASE_1
	picture = GFX_evt_khemri_king
	min_age = 16
	only_playable = yes

	trigger = {
		is_disease_immune_race = no
		not = {
			religion = nehekharan_gods
			trait = has_tomb_rot
		}
		location = {
			region = world_nehekhara
		}
	}

	mean_time_to_happen = {
		years = 20
		modifier = {
			factor = 0.5
			trait = greedy
		}
		modifier = {
			factor = 0.025
			OR = {
				religion = nurgle
				society_member_of = cult_of_nurgle
			}
		}
	}

	option = {
		name = EVTOPT_DISEASE_1A
		add_trait = has_tomb_rot
	}
}

character_event = {
	id = disease.2
	desc = EVTDESC_DISEASE_2
	picture = GFX_evt_recovery

	trigger = {
		trait = has_creeping_buboes
		nor = {
			religion = nurgle
			society_member_of = cult_of_nurgle
		}
	}

	mean_time_to_happen = {
		months = 2
		modifier = {
			factor = 2.0
			trait = ill
		}
		modifier = {
			factor = 4.0
			trait = infirm
		}
		modifier = {
			factor = 0.75
			base_health = 5.1
		}
		modifier = {
			factor = 0.5
			base_health = 6
		}
		modifier = {
			factor = 0.5
			base_health = 7
		}
		modifier = {
			factor = 1.5
			NOT = {
				base_health = 5
			}
		}
		modifier = {
			factor = 5.0
			trait = incapable
		}
		modifier = {
			factor = 0.75
			NOT = {
				age = 40
			}
		}
		modifier = {
			factor = 0.75
			NOT = {
				age = 20
			}
		}
		modifier = {
			factor = 2.0
			OR = {
				age = 60
				NOT = {
					age = 10
				}
			}
		}
	}

	option = {
		name = EVTOPT_DISEASE_1A
		remove_trait = has_creeping_buboes
	}
}
