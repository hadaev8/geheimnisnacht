		event_target:magic_caster = {
			clr_character_flag = casting
		}
	}
	option = {
	trigger = {event_target:magic_caster = { not = { check_variable = { which = "spell_power" value = 0}} }}
		name = high_magic_lore.1c
		event_target:magic_caster = {
			clr_character_flag = casting
		}
	}

}

character_event = {
	id = high_magic_lore.100#"High Magic; Soul Quench"
	desc = high_magic_lore.100_desc#"You channel the arcane powers and lash out towards the enemy."
	is_triggered_only = yes
	picture = GFX_evt_mage_lore_high_magic
	immediate = {
		event_target:magic_caster = {
			change_variable = {which = "spell_power" value = -1}
			miscast_test_1 = yes
		}
	}
	option = {
		name = high_magic_lore.100a#
		dispel_setup_effect = yes
		if = {
			limit = {
				event_target:magic_caster = {
					has_character_flag = has_dispel
				}
			}
			dispel_saving_roll_40 = yes
			event_target:magic_caster = {
				clr_character_flag = has_dispel
			}
		}
		if = {
			limit = {
				event_target:magic_caster = {
					NOR = {
						has_character_flag = miscast
						has_character_flag = dispeled
					}
				}
			}
			character_event = { id = high_magic_lore.101}
		}
		if = {
			limit = {
				event_target:magic_caster = {
					has_character_flag = dispeled
				}
			}
			event_target:magic_caster = { clr_character_flag = dispeled}
			character_event = { id = high_magic_lore.501}
		}
		if = {
			limit = {
				event_target:magic_caster = { has_character_flag = miscast }
			}
			character_event = {id = magic_system.99991}#If has the miscast flag, skip to the miscast event table
		}

	}
}


character_event = {
	id = high_magic_lore.101#Drain Magic Power 1"
	desc = high_magic_lore.101_desc#"Your spell burns through the air, taking from all enemies the ability to cast. "
	is_triggered_only = yes
	picture = GFX_evt_mage_lore_high_magic
	immediate = {
		any_character = {
			limit = {
				war_with = root
				at_location = event_target:magic_caster
				or = {
					has_character_modifier = magic_power_1
					has_character_modifier = magic_power_2
					has_character_modifier = magic_power_3
					has_character_modifier = magic_power_4
				}
				not = {
					has_character_modifier = magic_cooldown
				}
			}
			if = {
				limit = {
					not = {
						has_character_modifier = magic_drain_1
						has_character_modifier = magic_drain_2
						has_character_modifier = magic_drain_3
						has_character_modifier = magic_drain_4
					}
				}
				set_character_flag = mdrain
				add_character_modifier = { modifier = magic_drain_1 duration = 200 }
			}

			if = {
				limit = {
					has_character_modifier = magic_drain_1
					has_character_modifier = magic_power_1
				}
				add_character_modifier = { modifier = magic_cooldown duration = 300 }
				remove_character_modifier  = magic_drain_1
			}

			if = {
				limit ={
					not = {
						has_character_flag = mdrain
						has_character_modifier = magic_drain_2
						has_character_modifier = magic_drain_3
						has_character_modifier = magic_drain_4
					}
					has_character_modifier = magic_drain_1
				}
				remove_character_modifier = magic_drain_1
				set_character_flag = mdrain
				add_character_modifier = { modifier = magic_drain_2 duration = 200 }
			}
			if = {
				limit = {
					has_character_modifier = magic_drain_2
					has_character_modifier = magic_power_2
				}
				add_character_modifier = { modifier = magic_cooldown duration = 300 }
				remove_character_modifier  = magic_drain_2
			}

			if = {
				limit = {
					not = {
						has_character_flag = mdrain
						has_character_modifier = magic_drain_1
						has_character_modifier = magic_drain_3
						has_character_modifier = magic_drain_4
					}
					has_character_modifier = magic_drain_2
				}
				remove_character_modifier  = magic_drain_2
				set_character_flag = mdrain
				add_character_modifier = { modifier = magic_drain_3 duration = 200 }
			}
			if = {
				limit = {
					has_character_modifier = magic_drain_3
					has_character_modifier = magic_power_3
				}
				add_character_modifier = { modifier = magic_cooldown duration = 300 }
				remove_character_modifier  = magic_drain_3
			}

			if = {
				limit ={
					not = {
						has_character_flag = mdrain
						has_character_modifier = magic_drain_1
						has_character_modifier = magic_drain_2
						has_character_modifier = magic_drain_4
					}
					has_character_modifier = magic_drain_3
				}
				remove_character_modifier  = magic_drain_3
				set_character_flag = mdrain
				add_character_modifier = { modifier = magic_drain_4 duration = 200 }
			}
			if = {
				limit = {
					has_character_modifier = magic_drain_4
					has_character_modifier = magic_power_4
				}
				add_character_modifier = { modifier = magic_cooldown duration = 300 }
				remove_character_modifier = magic_drain_4
			}
		clr_character_flag = mdrain
		}
	}

	option = {
		name = high_magic_lore.101a#No
		character_event = {id = high_magic_lore.500}

	}
}

character_event = {
	id = high_magic_lore.402#Firey_convocation event 1"
	#desc = high_magic_lore.402_desc#"Your power consumes the life of your chosen target "
	is_triggered_only = yes
	hide_window = yes
	#picture = GFX_evt_mage_lore_high_magic
	immediate = {
		if = {
			limit = {
				event_target:magic_caster = { check_variable = {which = "fieryconv" value = 0}}
			}
			event_target:magic_caster = { change_variable = {which = "fieryconv" value = -1}}
			any_character = {
				limit = {
					war_with = root
					any_army = {
						location = { province = event_target:spellloc }
					}
				}
				random_army = {
					limit = {
						location = { province = event_target:spellloc }

					}
					troops = -0.02
					morale = -0.02
				}
			}
			any_character = {
				if = {
					limit = {
						or = {
							is_vassal_or_below = root
							character = root
						}
						at_location = event_target:spellloc
					}
					siege = {
						enemy = {
							troops = -0.02
						}
					}
				}
			}
			repeat_event = { id = high_magic_lore.402 days = 1}
		}
	}
}
