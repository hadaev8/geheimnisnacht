long_character_event = {
	id = fire_lore.1#"High Magic"
	desc = fire_lore.1_desc#"High Magic, the true magic. "
	is_triggered_only = yes

	picture = GFX_evt_mage_lore_fire
	immediate = {
		event_target:magic_caster = {
			set_variable = {which = "fierystorm" value = 0}
			magic_xp_counter = yes
			magic_points_setup = yes
			magic_spells_known = yes
		}
	}
	option = {
	trigger = {event_target:magic_caster = {check_variable = { which = "spell_power" value = 1}} }
		name = fire_lore.1a#"High Magic; Drain Magic"
		long_character_event = {id = fire_lore.2}
	}
	option = {
	trigger = {event_target:magic_caster = {check_variable = { which = "spell_power" value = 1}} }
		name = fire_lore.1b#"High Magic; Drain Magic"
		event_target:magic_caster = {
			clr_character_flag = casting
		}
	}
	option = {
	trigger = {event_target:magic_caster = { not = { check_variable = { which = "spell_power" value = 1}} }}
		name = fire_lore.1c#"High Magic; Drain Magic"
		event_target:magic_caster = {
			clr_character_flag = casting
		}
	}

}

character_event = {
	id = fire_lore.100#"High Magic; Soul Quench"
	desc = fire_lore.100_desc#"You channel the arcane powers and lash out towards the enemy."
	is_triggered_only = yes
	picture = GFX_evt_mage_lore_fire
	immediate = {
		event_target:magic_caster = {
			change_variable = {which = "spell_power" value = -1}
			miscast_test_5 = yes
		}
	}
	option = {
		name = fire_lore.100a#
		dispel_setup_effect = yes
		if = {
			limit = {
				event_target:magic_caster = {
					has_character_flag = has_dispel
				}
			}
			dispel_saving_roll_50 = yes
			event_target:magic_caster = {
				clr_character_flag = has_dispel
			}
		}
		if = {
			limit = {
				event_target:magic_caster = {
					NOR = {
						has_character_flag = miscast
						has_character_flag = dispeled
					}
				}
			}
			character_event = { id = fire_lore.101}
		}
		if = {
			limit = {
				event_target:magic_caster = {
					has_character_flag = dispeled
				}
			}
			event_target:magic_caster = { clr_character_flag = dispeled}
			character_event = { id = fire_lore.501}
		}
		if = {
			limit = {
				event_target:magic_caster = { has_character_flag = miscast }
			}
			character_event = {id = magic_system.99991}#If has the miscast flag, skip to the miscast event table
		}

	}
}
